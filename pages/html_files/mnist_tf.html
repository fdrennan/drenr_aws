def logit(z):
    return 1 / (1 + np.exp(-z))

def leaky_relu(z, name=None):
    return tf.maximum(0.01 * z, z, name=name)

def elu(z, alpha=1):
    return np.where(z < 0, alpha * (np.exp(z) - 1), z)

# https://arxiv.org/pdf/1706.02515.pdf
# It outperforms the other activation functions very significantly for deep neural networks

def selu(z,
         scale=1.0507009873554804934193349852946,
         alpha=1.6732632423543772848170429916717):
    return scale * tf.where(z >= 0.0, z, alpha * tf.nn.elu(z))

In [10]:
with tf.name_scope("dnn"):
    
    hidden_layers = tf.layers.dense(inputs     = X, 
                               units      = LAYERS[0], 
                               activation = ACTIVATION, 
                               name       = "hidden_init")
    
    for layer in np.arange(1, len(LAYERS)):
           hidden_layers = tf.layers.dense(inputs     = hidden_layers, 
                                           units      = LAYERS[layer], 
                                           activation = ACTIVATION, 
                                           name       = "hidden_" + str(layer))
    
    logits   = tf.layers.dense(inputs     = hidden_layers, 
                               units      = N_OUTPUTS,
                               name       = "outputs")
In [11]:
with tf.name_scope("loss"):
    
    xentropy = tf.nn.sparse_softmax_cross_entropy_with_logits(labels = y, 
                                                              logits = logits)
    loss     = tf.reduce_mean(input_tensor = xentropy, 
                              name         = "loss")
In [12]:
with tf.name_scope("train"):

    training_op = optimizer.minimize(loss)
In [13]:
with tf.name_scope("eval"):
    
    # This outputs a `batch_size` bool array, an entry `out[i]` is `true` if the
    # prediction for the target class is among the top `k` predictions among
    # all predictions for example `i`.
    # result -> [False True True True False False True True False True True True]
    correct = tf.nn.in_top_k(predictions = logits, 
                             targets     = y, 
                             k           = 1)
    
    # Example on how cast works.
    # x = tf.constant([1.8, 2.2], dtype=tf.float32)
    # tf.cast(x, tf.int32)  result -> # [1, 2], dtype=tf.int32
    accuracy = tf.reduce_mean(tf.cast(x = correct, dtype = tf.float32))

with tf.Session() as sess:
    
    init.run()
    
    validation_feeder = {X: mnist.validation.images, 
                         y: mnist.validation.labels}
    
    test_feeder       = {X: mnist.test.images, 
                         y: mnist.test.labels}
    
    for epoch in range(N_EPOCHS):
        
        for iteration in range(mnist.train.num_examples // BATCH_SIZE):
            
            X_batch, y_batch = mnist.train.next_batch(BATCH_SIZE)
            
            train_feeder = {X: X_batch, 
                            y: y_batch}
            
            sess.run(fetches   = training_op, 
                     feed_dict = train_feeder)
            
        if epoch % 5 == 0:
            
            acc_train = accuracy.eval(feed_dict = train_feeder)
            
            acc_validation  = accuracy.eval(feed_dict = validation_feeder)
            
            # Dictionary to average all validation scores.
            accuracies_validation.append(acc_validation)
            
            print(epoch, "Batch accuracy:", acc_train, "Validation accuracy:", acc_validation)
    
    test_accuracy  = accuracy.eval(feed_dict = test_feeder)  
    print("Test Accuracy: ", test_accuracy)
    
    results = sess.run(logits, feed_dict = test_feeder)
    
    save_path = saver.save(sess, "./my_model_final.ckpt")
