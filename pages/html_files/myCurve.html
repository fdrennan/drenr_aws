curve_fitter <- function(expr="", columns = "c(x, y)", dataFrame = "dataframe") {
  x <- dataFrame[[columns[[1]]]]
  y <- dataFrame[[columns[[2]]]]
  expr <- unlist(strsplit(expr,","))
  nCol <- length(expr) + 1
  ex <- rep("rep(1,length(x))", nCol)
  ex[1:length(expr)] <- expr
  A <- sapply(1:length(ex), function(i) eval(parse(text=ex[i])))
  AtA <- t(A)%*%A
  B <- t(A) %*% y
  coeff <- solve(AtA)%*%B
  #coeff <- c(curve_fit[length(curve_fit),], curve_fit[1:length(curve_fit) - 1,])
  coeff <- c(coeff[length(coeff)], coeff[1:length(coeff) - 1])
  coeff
}

With linear data

x <- 1:10 
y <- 1:10
my_data <- data.frame("x" = x, "y" = y)

curve_coefficients <- curve_fitter("x", c(1, 2), my_data)
plot(my_data$x, my_data$y)
curve(curve_coefficients[1] + curve_coefficients[2]*x, add = TRUE)

With sinusoidal data

x <- 1:25 
y <- sin(1:25) + 1:25
my_data <- data.frame("x" = x, "y" = y)

curve_coefficients <- curve_fitter("x, sin(x)", c(1, 2), my_data)
plot(my_data$x, my_data$y)
curve(curve_coefficients[1] + curve_coefficients[2]*x + curve_coefficients[3]*sin(x), add = TRUE)
