%pylab inline
pylab.rcParams['figure.figsize'] = (8, 5)

import numpy as np
import matplotlib.pyplot as plt
t = np.r_[0:10:0.1]
freq = 0.5
x = np.sin(2*np.pi*freq*t)
plt.plot(t,x)
plt.xlabel('Time[sec]')
plt.ylabel('Values')
plt.show()

freq = 0.5
