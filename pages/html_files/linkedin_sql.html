Slightly more complicated query.

query <- '
SELECT 
  COUNT(*),
  year,
  month,
  day
FROM 
  p_flights
GROUP BY 
  year, 
  month,
  day
ORDER BY year, month, day
'

results <- tbl(con, sql(query)) %>% as.data.frame()


results %>% 
  mutate(date = as.Date(paste0(year, "-", month, "-", day))) %>% 
  ggplot() +
  aes(x = date, y = count) +
  geom_line()
